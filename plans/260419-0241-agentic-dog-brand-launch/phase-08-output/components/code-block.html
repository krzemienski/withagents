The agent-contracts library exposes a typed AgentInterface that both the orchestrator and any downstream trace-timeline consumer can import. Run pnpm add @withagents/agent-contracts to get started.

Fenced — TypeScript, copy button
typescript 
import { defineContract } from '@withagents/agent-contracts';

const runbookContract = defineContract({
  name: 'triage-runbook',
  version: '1.0.0',
  tools: ['read_file', 'search_logs', 'create_ticket'],
  maxRounds: 12,
  outputSchema: TriageResult,
});

// Export for orchestrator consumption
export default runbookContract;

from dataclasses import dataclass, field
from typing import Any, Dict, Optional

@dataclass
class ContextLayer:
    session_id: str
    observations: Dict[str, Any] = field(default_factory=dict)
    parent_id: Optional[str] = None

    def observe(self, key: str, value: Any) -> None:
        self.observations[key] = value  # always mutable
Highlighted line: the observation write is intentionally mutable — see memory-layer docs.
